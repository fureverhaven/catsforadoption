        }

        .toggle-btn span {
            display: block;
            width: 30px;
            height: 5px;
            background-color: #333;
            margin: 6px 0;
            transition: 0.3s;
        }

        /* When sidebar is open */

            left: 0;
        }

        /* Main content area */
        .content {
            margin-left: 0;
            padding: 20px;
            transition: margin-left 0.3s;
        }

        .content.shifted {
            margin-left: 250px;
        }

    </style>
</head>
<body>

    <!-- Toggle Button -->
    <button class="toggle-btn" onclick="toggleSidebar()">
        <span></span>
        <span></span>
        <span></span>
    </button>

    <!-- Sidebar -->
    <div class="sidebar">
        <a href="#">Home</a>
        <a href="#">About</a>
        <a href="#">Services</a>
        <a href="#">Portfolio</a>
        <a href="#">Contact</a>
    </div>

    <!-- Main Content -->
    <div class="content">
        <h1>Welcome to the Website</h1>
        <p>This is a sidebar navigation menu with an animated toggle bar.</p>
    </div>

    <script>
        // Function to toggle sidebar open/close
        function toggleSidebar() {
            const sidebar = document.querySelector('.sidebar');
            const content = document.querySelector('.content');
            sidebar.classList.toggle('open');
            content.classList.toggle('shifted');
        }
    </script>

</body>

    • The function toggleSidebar() is used to add or remove the .open class from the sidebar and the .shifted class from the content, triggering the opening and closing animation.

  • Initially, the sidebar is hidden on the left side of the screen.
